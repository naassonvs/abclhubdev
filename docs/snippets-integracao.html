// ── Configuração da página ──────────────────────────────────────────────────
const MODULO = 'vendinha'; // ou 'livraria'

let formaPgtoSelecionada = 'dinheiro';

function selecionarPgto(btn) {
  document.querySelectorAll('.btn-pgto').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  formaPgtoSelecionada = btn.dataset.pgto;
}

function toggleOferta() {
  const chk = document.getElementById('chk-oferta');
  document.getElementById('detalhe-oferta').style.display = chk.checked ? 'block' : 'none';
}

function recalcularTotal() {
  // Adapte para o seu sistema de carrinho existente
  const subtotal  = calcularSubtotal(); // sua função existente
  const desconto  = parseInt(document.getElementById('campo-desconto').value) || 0;
  const total     = Math.max(0, subtotal - desconto);
  // Exibir total — adapte ao seu elemento
  document.getElementById('valor-total').textContent = 'R$ ' + total.toFixed(2).replace('.',',');
  return { subtotal, desconto, total };
}

// ── Confirmar venda com Firebase ────────────────────────────────────────────
async function confirmarVendaComFirebase() {
  const btn = document.querySelector('[onclick="confirmarVendaComFirebase()"]');

  if (!carrinho || carrinho.length === 0) {
    alert('Carrinho vazio!');
    return;
  }

  btn.disabled    = true;
  btn.textContent = '⏳ Salvando...';

  const { subtotal, desconto, total } = recalcularTotal();
  const chkOferta  = document.getElementById('chk-oferta');
  const ehOferta   = chkOferta ? chkOferta.checked : false;
  const valorOferta = ehOferta
    ? parseInt(document.getElementById('campo-valor-oferta').value) || 0
    : null;

  // Monte os produtos com o schema esperado
  const produtos = carrinho.map(item => ({
    id:             item.id   || null,
    nome:           item.nome,
    quantidade:     parseInt(item.qtd),
    preco_unitario: item.preco,
    preco_total:    item.preco * parseInt(item.qtd),
  }));

  const venda = {
    tipo:             MODULO,
    produtos,
    subtotal,
    desconto,
    total,
    eh_oferta:        ehOferta,
    valor_oferta:     valorOferta,
    status_oferta:    ehOferta ? 'pendente' : null,
    forma_pagamento:  ['credito','debito'].includes(formaPgtoSelecionada) ? 'cartao' : formaPgtoSelecionada,
    cartao_tipo:      ['credito','debito'].includes(formaPgtoSelecionada) ? formaPgtoSelecionada : null,
    operador_uid:     null,     // substitua por firebase.auth().currentUser?.uid se usar auth
    operador_nome:    'Operador', // substitua pelo nome real
    ano_acampamento:  new Date().getFullYear(),
  };

  try {
    if (navigator.onLine) {
      // Online: salva direto
      await VendasFirebase.salvarVendaFirebase(venda, MODULO);
      mostrarToast('✅ Venda salva!');
    } else {
      // Offline: enfileira
      await OfflineSync.adicionarFilaSincronizacao(venda);
      mostrarToast('📴 Offline — venda salva localmente');
    }

    limparCarrinho(); // sua função existente
    document.getElementById('campo-desconto').value = 0;
    if (document.getElementById('chk-oferta')) {
      document.getElementById('chk-oferta').checked = false;
      toggleOferta();
    }
    formaPgtoSelecionada = 'dinheiro';
    document.querySelectorAll('.btn-pgto').forEach(b => b.classList.remove('active'));
    document.querySelector('[data-pgto="dinheiro"]').classList.add('active');

  } catch (err) {
    console.error('[PDV] Erro ao salvar venda:', err);
    // Fallback para fila offline
    try {
      await OfflineSync.adicionarFilaSincronizacao(venda);
      mostrarToast('⚠️ Erro na rede — venda salva offline');
    } catch (e2) {
      alert('Erro crítico ao salvar a venda. Anote manualmente.');
    }
  }

  btn.disabled    = false;
  btn.textContent = '✅ Confirmar Venda';
}

// ── Editar/Cancelar venda ────────────────────────────────────────────────────
async function editarVenda(vendaId, novosDados) {
  try {
    await VendasFirebase.editarVendaFirebase(vendaId, novosDados);
    mostrarToast('✏️ Venda atualizada!');
  } catch (err) {
    alert('Erro ao editar: ' + err.message);
  }
}

async function cancelarVenda(vendaId) {
  if (!confirm('Cancelar esta venda? O estoque será restaurado.')) return;
  try {
    await VendasFirebase.cancelarVendaFirebase(vendaId, 'Cancelado pelo operador');
    mostrarToast('🚫 Venda cancelada');
  } catch (err) {
    alert('Erro ao cancelar: ' + err.message);
  }
}

// Busca produtos da coleção 'loja_estoque' no Firestore
async function carregarProdutos() {
  const container = document.getElementById('produtos-grid'); // seu elemento
  container.innerHTML = '<p style="color:var(--suave)">Carregando...</p>';

  try {
    const snap = await firebase.firestore()
      .collection('loja_estoque')
      .where('ativo', '==', true)
      .orderBy('nome')
      .get();

    if (snap.empty) {
      container.innerHTML = '<p>Nenhum produto disponível.</p>';
      return;
    }

    const produtos = snap.docs.map(d => ({ id: d.id, ...d.data() }));
    renderProdutos(produtos); // sua função de renderização

    // Salva localmente para acesso offline
    localStorage.setItem('abcl_loja_produtos', JSON.stringify(produtos));

  } catch (err) {
    console.warn('[Loja] Firestore indisponível, usando cache local:', err);
    const cache = JSON.parse(localStorage.getItem('abcl_loja_produtos') || '[]');
    if (cache.length > 0) {
      renderProdutos(cache);
    } else {
      container.innerHTML = '<p>Sem conexão e sem cache disponível.</p>';
    }
  }
}

// Chamar no DOMContentLoaded:
document.addEventListener('DOMContentLoaded', carregarProdutos);

async function finalizarPedido() {
  if (!carrinho || carrinho.length === 0) { alert('Carrinho vazio!'); return; }

  const pedido = {
    tipo:            'loja',
    produtos:        carrinho.map(i => ({

      nome:           i.nome,
      quantidade:     parseInt(i.qtd),
      preco_unitario: i.preco,
      preco_total:    i.preco * parseInt(i.qtd),
    })),
    subtotal:        calcularSubtotal(),
    desconto:        0,
    total:           calcularSubtotal(),
    forma_pagamento: 'pix', // loja usa PIX por padrão
    operador_uid:    null,
    operador_nome:   'Cliente',
    ano_acampamento: new Date().getFullYear(),
  };

  try {
    if (navigator.onLine) {
      await VendasFirebase.salvarVendaFirebase(pedido, 'loja');
    } else {
      await OfflineSync.adicionarFilaSincronizacao(pedido);

}

// ══════════════════════════════════════════════════════
// PDV — Integração Firebase no financeiro.html
// ══════════════════════════════════════════════════════

let _vendaPDVSelecionada = null;
let _unsubPDV = null; // unsubscribe do listener em tempo real

const fmtR = v => 'R$ ' + Number(v || 0).toFixed(2).replace('.', ',');

const MODULO_LABEL = { vendinha: 'Vendinha', livraria: 'Livraria', loja: 'Loja' };
const MODULO_COR   = { vendinha: 'var(--verde-cl)', livraria: 'var(--azul)', loja: 'var(--amarelo)' };

// Carrega resumo em tempo real ao abrir a tab PDV
function initPDVTab() {
  // Escuta resumo em tempo real
  VendasFirebase.escutarResumoFinanceiro(new Date().getFullYear(), (resumo) => {
    document.getElementById('pdv-total-vendinha').textContent = fmtR(resumo.vendinha);
    document.getElementById('pdv-total-livraria').textContent = fmtR(resumo.livraria);
    document.getElementById('pdv-total-loja').textContent     = fmtR(resumo.loja);
    document.getElementById('pdv-total-geral').textContent    = fmtR(resumo.total);
  });

  carregarVendasPDV();
}

async function carregarVendasPDV() {
  const modulo = document.getElementById('pdv-filtro-modulo').value;
  const data   = document.getElementById('pdv-filtro-data').value;
  const status = document.getElementById('pdv-filtro-status').value;
  const loading = document.getElementById('pdv-loading');
  const tbody   = document.getElementById('pdv-tabela-vendas');

  loading.textContent = 'carregando...';
  tbody.innerHTML = '';

  // Cancela listener anterior
  if (_unsubPDV) { _unsubPDV(); _unsubPDV = null; }

  // Listener em tempo real
  _unsubPDV = VendasFirebase.escutarVendasModulo(modulo || null, (vendas) => {
    // Aplica filtros locais (data e status, pois WHERE composto exige índice)
    let filtradas = vendas;
    if (data)   filtradas = filtradas.filter(v => v.data === data);
    if (status) filtradas = filtradas.filter(v => v.status === status);

    loading.textContent = filtradas.length + ' venda(s)';
    renderTabelaVendasPDV(filtradas);
  });
}

function renderTabelaVendasPDV(vendas) {
  const tbody = document.getElementById('pdv-tabela-vendas');

  if (!vendas.length) {
    tbody.innerHTML = '<tr><td colspan="8" style="text-align:center;color:var(--suave);padding:1.5rem;font-size:.78rem;">Nenhuma venda encontrada.</td></tr>';
    return;
  }

  tbody.innerHTML = vendas.map(v => {
    const cor   = MODULO_COR[v.tipo]   || 'var(--suave)';
    const label = MODULO_LABEL[v.tipo] || v.tipo;
    const prods = (v.produtos || []).map(p => p.nome + ' x' + p.quantidade).join(', ') || '—';
    const pgto  = v.forma_pagamento === 'cartao'
      ? (v.cartao_tipo === 'credito' ? '💳 Crédito' : '💳 Débito')
      : v.forma_pagamento === 'pix' ? '📲 PIX' : '💵 Dinheiro';
    const statusTag = v.status === 'cancelada'
      ? '<span class="td-tag tag-despesa">Cancelada</span>'
      : '<span class="td-tag tag-receita">Confirmada</span>';
    const ofertaTag = v.eh_oferta
      ? '<span class="td-tag tag-pendente">🎁 ' + (v.status_oferta || 'oferta') + '</span>'
      : '—';

    return `<tr style="cursor:pointer;" onclick="abrirModalVenda('${v.id}')">
      <td style="font-size:.75rem;">${v.data || '—'} ${v.hora || ''}</td>
      <td><span style="color:${cor};font-weight:600;font-size:.8rem;">${label}</span></td>
      <td style="font-size:.72rem;color:var(--suave);max-width:180px;overflow:hidden;text-overflow:ellipsis;white-space:nowrap;" title="${prods}">${prods}</td>
      <td style="font-size:.75rem;">${pgto}</td>
      <td>${ofertaTag}</td>
      <td class="td-valor" style="color:${v.status === 'cancelada' ? 'var(--suave)' : 'var(--verde-cl)'};text-decoration:${v.status === 'cancelada' ? 'line-through' : 'none'};">${fmtR(v.total)}</td>
      <td>${statusTag}</td>
      <td><button class="btn btn-ghost btn-sm" onclick="event.stopPropagation();abrirModalVenda('${v.id}')">Ver</button></td>
    </tr>`;
  }).join('');

  // Guarda vendas para o modal
  window._pdvVendas = vendas;
}

        <span>Desconto</span><span>- ${fmtR(venda.desconto)}</span></div>
      <div style="display:flex;justify-content:space-between;font-weight:700;font-size:.95rem;">
        <span>Total</span><span style="color:var(--verde-cl);">${fmtR(venda.total)}</span></div>
    </div>
    <div style="margin-top:.65rem;font-size:.72rem;color:var(--suave);">
      Pagamento: <strong style="color:var(--texto);">${venda.forma_pagamento}</strong>
      ${venda.cartao_tipo ? ' (' + venda.cartao_tipo + ')' : ''}
       |  Operador: <strong style="color:var(--texto);">${venda.operador_nome || '—'}</strong>
      ${venda.eh_oferta ? '<br>🎁 Oferta: <strong style="color:#fbbf24;">' + fmtR(venda.valor_oferta) + '</strong> — status: ' + (venda.status_oferta || '—') : ''}
    </div>`;

  const btnCanc = document.getElementById('pdv-btn-cancelar-venda');
  btnCanc.style.display = venda.status === 'cancelada' ? 'none' : 'inline-flex';
  btnCanc.onclick = () => cancelarVendaPDV(venda.id);

  const overlay = document.getElementById('pdv-modal-overlay');
  overlay.style.display = 'flex';
}

function fecharModalVenda() {
  document.getElementById('pdv-modal-overlay').style.display = 'none';
  _vendaPDVSelecionada = null;
}

async function cancelarVendaPDV(id) {
  if (!confirm('Cancelar esta venda? O estoque será restaurado.')) return;
  try {
    await VendasFirebase.cancelarVendaFirebase(id, 'Cancelado pelo financeiro');
    fecharModalVenda();
    toast('🚫 Venda cancelada com sucesso');
  } catch (err) {
    alert('Erro ao cancelar: ' + err.message);
  }
}

}
